subcheck =
  type: 'string'
  match: ///
    ^
    [a-zA-Z0-9]
    |[a-zA-Z0-9][a-zA-Z0-9\-_]{0,61}[a-zA-Z0-9]

&nbsp;
# Type implementation
# -------------------------------------------------
exports.describe = (work, cb) -&gt;
  text = 'A valid hostname. '
  text += check.optional.describe work
  text = text.replace /\. It's/, ' which is'
  # subcheck
  name = work.spec.name ? 'value'
  if work.path.length
    name += "/#{work.path.join '/'}"
  check.describe
    name: name
    schema: subcheck
  , (err, subtext) -&gt;
    # no error in string describe possible, so go on
    cb null, text + subtext
&nbsp;
exports.run = (work, cb) -&gt;
  debug "#{work.debug} with #{util.inspect work.value} as #{work.pos.type}"
  debug "#{work.debug} #{chalk.grey util.inspect work.pos}"
  # base checks
  try
    return cb() if check.optional.run work
  catch err
    return work.report err, cb
  # validate using subcheck
  name = work.spec.name ? 'value'
  if work.path.length
    name += "/#{work.path.join '/'}"
  check.run
    name: name
    value: work.value
    schema: subcheck
  , cb
&nbsp;
exports.selfcheck = (schema, cb) -&gt;
  check.run
    schema:
      type: 'object'
      allowedKeys: true
      keys: object.extend {}, check.base,
        default: subcheck
    value: schema
  , cb
