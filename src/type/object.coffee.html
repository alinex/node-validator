  • instanceOf - only objects of given class type are allowed
    • 

  • mandatoryKeys - the list of elements which are mandatory
    • 

  • allowedKeys - gives a list of elements which are also allowed
or true to use the list from entries definition

exports.selfcheck = (schema, cb) ->
  check.run
    schema:
      type: 'object'
      allowedKeys: true
      keys: object.extend {}, check.base,
        default:
          type: 'object'
          optional: true
        instanceOf:
          type: 'function'
          optional: true
        mandatoryKeys:
          type: 'or'
          optional: true
          or: [
            type: 'boolean'
          ,
            type: 'array'
            entries:
              type: 'or'
              or: [
                type: 'string'
              ,
                type: 'regexp'
              ]
          ]
        allowedKeys:
          type: 'or'
          optional: true
          or: [
            type: 'boolean'
          ,
            type: 'array'
            entries:
              type: 'or'
              or: [
                type: 'string'
              ,
                type: 'regexp'
              ]
          ]
        entries:
          type: 'array'
          optional: true
        keys:
          type: 'object'
          optional: true
    value: schema
  , cb
