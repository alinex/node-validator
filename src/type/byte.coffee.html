    • 

  • max - (integer) the biggest allowed number

    Selfcheck
  




        		
        
  selfcheck: (name, options) ->
    validator = require '../index'
    validator.check name,
      type: 'object'
      allowedKeys: true
      entries:
        type:
          type: 'string'
        title:
          type: 'string'
          optional: true
        description:
          type: 'string'
          optional: true
        optional:
          type: 'boolean'
          optional: true
        default:
          type: 'integer'
          optional: true
        unit:
          type: 'string'
          default: 'B'
          matches: /^[kMGTPEZY]?([bB]|bps)$/
        min:
          type: 'any'
          optional: true
          entries: [
            type: 'integer'
            min: 0
          ,
            rules.selfcheck.reference
          ]
        max:
          type: 'any'
          optional: true
          min:
            reference: 'relative'
            source: '<min'
          entries: [
            type: 'integer'
          ,
            rules.selfcheck.reference
          ]
    , options

optimize = (options) ->
  if options.decimals and not options.round?
    options.round = true
  if options.round and not options.decimals?
    options.decimals = 0
  unless options.min
    options.min = 0
  options
