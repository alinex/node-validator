  ¶



run the check

        		
        
  debug "#{spec.debug} check as #{spec.schema.title ? spec.schema.type}"
  check.run spec, (err, result) ->
    if err
      debug "#{spec.debug} failed with: #{err.message}"
    else
      debug "#{spec.debug} succeeded with: #{util.inspect result}"
    cb err, result
